from scipy.stats import geom
import numpy as np
import matplotlib.pyplot as plt
fig, ax = plt.subplots(1, 1)

p = 0.5
X = geom(p)
x = np.arange(0, 5)
ax.plot(x, geom.pmf(x, p), 'bo', ms=8, label='geom pmf')
ax.vlines(x, 0, geom.pmf(x, p), colors='b', lw=5, alpha=0.5)
